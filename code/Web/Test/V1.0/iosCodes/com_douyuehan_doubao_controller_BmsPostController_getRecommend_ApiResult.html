#import <Foundation/Foundation>;

@interface ApiResult : NSObject

//业务状态码
@property (nonatomic, strong) long code;
//结果集
@property (nonatomic, strong) BmsPost[] data;
//接口描述
@property (nonatomic, assign) NSString message;


@end



#import <Foundation/Foundation>;

@interface BmsPost : NSObject

//主键
@property (nonatomic, assign) NSString id;
//标题
@property (nonatomic, assign) NSString title;
//markdown
@property (nonatomic, assign) NSString content;
//作者ID
@property (nonatomic, assign) NSString userId;
//评论数
@property (nonatomic, strong) NSInteger comments;
//收藏数
@property (nonatomic, strong) NSInteger collects;
//浏览数
@property (nonatomic, strong) NSInteger view;
//专栏ID，默认不分栏
@property (nonatomic, strong) NSInteger sectionId;
//置顶
@property (nonatomic, strong) BOOL top;
//加精
@property (nonatomic, strong) BOOL essence;
//创建时间
@property (nonatomic, assign) NSString createTime;
//修改时间
@property (nonatomic, assign) NSString modifyTime;


@end
