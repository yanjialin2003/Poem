public class ApiResult implements Serializable{

    private long code; //业务状态码
    private BmsBillboard data; //结果集
    private String message; //接口描述

    public long getCode(){
        return code;
    }

    public void setCode(long code){
        this.code = code;
    }

    public BmsBillboard getData(){
        return data;
    }

    public void setData(BmsBillboard data){
        this.data = data;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

}

public class BmsBillboard implements Serializable{

    private int id; //主键
    private String content; //公告牌
    private String createTime; //公告时间
    private boolean show; //1：展示中，0：过期

    public int getId(){
        return id;
    }

    public void setId(int id){
        this.id = id;
    }

    public String getContent(){
        return content;
    }

    public void setContent(String content){
        this.content = content;
    }

    public String getCreateTime(){
        return createTime;
    }

    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

    public boolean getShow(){
        return show;
    }

    public void setShow(boolean show){
        this.show = show;
    }

}
